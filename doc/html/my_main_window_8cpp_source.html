510 void myMainWindow::runGlobalSmoothing()

516  float detail_preservation = static_cast<float>(slider_gs_detail_preservation->value());
517  smoothingTask->n_vertex_iterations = static_cast<int>(smoothness/30.0f + 7.0f);
518  smoothingTask->sigma_c_ratio = 0.5 + radius_ratio / 20.0f + smoothness/100.0f;
519  smoothingTask->sigma_s = 0.8f-detail_preservation * 0.006f;
520  cout << smoothingTask->sigma_s << endl;
521  smoothingTask->iteration_step_size = default_iteration_step_size;
522  smoothingTask->currentGlobalSmoothingIteration = 0;
523  smoothingTask->finalGlobalSmoothingIteration = smoothness_i / default_iteration_step_size;
524  smoothingTask->algorithm_flag = currentGlobalSmoothingAlgorithm;
525  smoothingThread.start();
526  setGlobalSmoothingStatus(gs_status_started);

529 void myMainWindow::updateGlobalSmoothing()
530 {

533  if (globalSmoothingStopped)

659 void myMainWindow::getSelection(vector<size_t> & selected_vertices_ids)
660 {
661  selected_vertices_ids.clear();
662  int k = slider_fs_radius->value();
663  glm::vec2 mousePos = output_mesh_visualizer_ptr->getCurrentMousePosition();
664  glm::vec3 dir = output_mesh_visualizer_ptr->getRayDirection(mousePos);
665  //cout << dir.x << " " << dir.y << " " << dir.z << endl;
666  vector<TriMesh::VertexHandle> selected_vertices;
667  glm::mat4 trMatrix = output_mesh_visualizer_ptr->getModelToWorldMatrix();
668  //compute neighborhood
669  float min_dist = 999999.0f;
670  TriMesh::VertexHandle nearest_vertex_handle;
671  for (TriMesh::FaceIter f_it = data.output_mesh.faces_begin(); f_it != data.output_mesh.faces_end(); f_it++)
672  {
673  glm::vec3 a, b, c;
674  TriMesh::VertexHandle a_vh, b_vh, c_vh;
675 
676  TriMesh::FaceVertexIter fv_it = data.output_mesh.fv_iter(*f_it);
677  a = glm::vec3(data.output_mesh.point(*fv_it)[0], data.output_mesh.point(*fv_it)[1], data.output_mesh.point(*fv_it)[2]);
678  a = glm::vec3(trMatrix*glm::vec4(a, 1.0));
679  a_vh = *fv_it;
680  fv_it++;
681  b = glm::vec3(data.output_mesh.point(*fv_it)[0], data.output_mesh.point(*fv_it)[1], data.output_mesh.point(*fv_it)[2]);
682  b = glm::vec3(trMatrix*glm::vec4(b, 1.0));
683  b_vh = *fv_it;
684  fv_it++;
685  c = glm::vec3(data.output_mesh.point(*fv_it)[0], data.output_mesh.point(*fv_it)[1], data.output_mesh.point(*fv_it)[2]);
686  c = glm::vec3(trMatrix*glm::vec4(c, 1.0));
687  c_vh = *fv_it;
688 
689  glm::vec3 ab = b - a;
690  glm::vec3 ac = c - a;
691 
692  glm::vec3 normal = glm::cross(ab, ac);
693 
694  glm::vec3 o = output_mesh_visualizer_ptr->getCamera()->getPosition();
695  float val = glm::dot(dir, normal);
696  if (val == 0.0f) continue;
697  float ti = glm::dot((a - o), normal) / val;
698  glm::vec3 temp = dir*ti;
699  glm::vec3 pi = o + temp;
700  glm::vec3 v0 = b - a;
701  glm::vec3 v1 = c - a;
702  glm::vec3 v2 = pi - a;
703  float d00 = glm::dot(v0, v0);
704  float d01 = glm::dot(v0, v1);
705  float d11 = glm::dot(v1, v1);
706  float d20 = glm::dot(v2, v0);
707  float d21 = glm::dot(v2, v1);
708  double denom = glm::dot(d00, d11) - glm::dot(d01, d01);
709  float v = (d11 * d20 - d01 * d21) / denom;
710  float w = (d00 * d21 - d01 * d20) / denom;
711  float u = 1.0 - v - w;
712 
713  bool liesOutside = v>1 || v<0 || w>1 || w<0 || u>1 || u<0;
714 
715  if (!liesOutside)
716  {
717  //cout << "intersect" << endl;
718  if (ti < min_dist && ti >= 0)
719  {
720  if (u>v)
721  {
722  if (u > w)
723  nearest_vertex_handle = a_vh;
724  else
725  nearest_vertex_handle = c_vh;
726  }
727  else if (v > w)
728  nearest_vertex_handle = b_vh;
729  else
730  nearest_vertex_handle = c_vh;
731  min_dist = ti;
732  }
733 
734  }
735  }
736  if (nearest_vertex_handle.is_valid())
737  {
738  getVertexNeighbors(data.output_mesh, nearest_vertex_handle, k, selected_vertices);
739  selected_vertices.push_back(nearest_vertex_handle);
740  }
741 
742  for (size_t i = 0; i < selected_vertices.size(); i++)
743  selected_vertices_ids.push_back(selected_vertices[i].idx());
744 }
